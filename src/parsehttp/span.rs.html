3/// A range and position of text within source text
4#[derive(Debug, Clone, Copy, PartialEq, Eq, Builder)]
5pub struct SpanPosition {
6    pub index: usize,
7    pub line: usize,
8    pub column: usize,
9}
10
11impl From<(usize, usize, usize)> for SpanPosition {
12    fn from(value: (usize, usize, usize)) -> Self {
13        Self::builder()
14            .index(value.0)
15            .line(value.1)
16            .column(value.2)
17            .build()
18    }
19}
20
21/// Construct a span position
22pub fn span_position(index: usize, line: usize, column: usize) -> SpanPosition {
23    (index, line, column).into()
24}

26/// A range and position of text within source text
27#[derive(Debug, Clone, Copy, PartialEq, Eq, Builder)]
28pub struct Span {
29    pub start: SpanPosition,
30    pub end: SpanPosition,
31}
32
33impl Span {
34    /// Get the slice of text the span represents in the source text
35    pub fn slice<'a>(&self, source: &'a str) -> &'a str {
36        &source[self.start.index..self.end.index]
37    }
38}
39
40/// Construct a span from two span positions
41pub fn span(start: impl Into<SpanPosition>, end: impl Into<SpanPosition>) -> Span {
42    Span {
43        start: start.into(),
44        end: end.into(),
45    }
46}
